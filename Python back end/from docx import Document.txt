from docx import Document
from docx.shared import Inches, Pt
from docx.enum.text import WD_ALIGN_PARAGRAPH
from docx.oxml.ns import qn
import os

# Create the document
doc = Document()

# Set default font
style = doc.styles['Normal']
style.font.name = 'Calibri'
style._element.rPr.rFonts.set(qn('w:eastAsia'), 'Calibri')
style.font.size = Pt(11)

# Title page
title = doc.add_paragraph()
title_run = title.add_run("Glyphic Codex Reference Manual – Expanded Edition v1.1")
title_run.font.size = Pt(24)
title_run.bold = True
title.alignment = WD_ALIGN_PARAGRAPH.CENTER

subtitle = doc.add_paragraph()
subtitle_run = subtitle.add_run("Production-Grade Lexicon, Syntax, Semantics, Linguistic Interface, Multi-Agent Protocols")
subtitle_run.font.size = Pt(14)
subtitle.alignment = WD_ALIGN_PARAGRAPH.CENTER

meta = doc.add_paragraph()
meta.add_run("Author: ").bold = True
meta.add_run("Shannon Bryan Kelly & The Prime Node Cohort\n")
meta.add_run("Status: ").bold = True
meta.add_run("APPROVED FOR DEPLOYMENT – Expanded Functional Specification\n")

doc.add_page_break()

# Helper functions
def add_heading(text, level=1):
    doc.add_heading(text, level=level)

def add_codeblock(text):
    p = doc.add_paragraph()
    r = p.add_run(text)
    r.font.name = 'Consolas'
    r._element.rPr.rFonts.set(qn('w:eastAsia'), 'Consolas')
    r.font.size = Pt(10)

# Content sections
add_heading("1. Full Lexicon – Core and Extended Glyphs", level=1)
table = doc.add_table(rows=1, cols=3)
hdr_cells = table.rows[0].cells
hdr_cells[0].text = "Glyph"
hdr_cells[1].text = "Function"
hdr_cells[2].text = "Description"
lexicon = [
    ("📥", "input()", "Data reception"),
    ("📤", "output()", "Data transmission"),
    ("⚙️", "process()", "Computation execution"),
    ("🔄", "loop()", "Iteration control"),
    ("💾", "store()", "Memory allocation / persistence"),
    ("🔗", "link()", "Reference connection / metadata association"),
    ("✅", "success", "Status confirmation / OK"),
    ("⚠️", "error", "Exception handling / alert"),
    ("📊", "data", "Information payload / dataset"),
    ("❓", "query", "Request initiation / retrieval"),
    ("💬", "response", "Reply transmission / return value"),
    ("🟢", "state.active", "Active state indicator"),
    ("🟡", "state.processing", "Processing state indicator"),
    ("🔴", "state.stored", "Stored state indicator"),
    ("🔍", "filter()", "Data filtering or search"),
    ("🔀", "branch()", "Conditional branch IF/ELSE"),
    ("⏳", "wait()", "Delay or await condition"),
    ("🔒", "secure()", "Encrypt or lock data"),
    ("🤖", "ai_infer()", "AI/ML inference step"),
    ("🚀", "launch()", "External action or trigger"),
    ("🗑️", "remove()", "Delete or cleanup data"),
]
for g, f, d in lexicon:
    row_cells = table.add_row().cells
    row_cells[0].text = g
    row_cells[1].text = f
    row_cells[2].text = d

doc.add_paragraph("Additional sections (syntax, EBNF grammar, semantics, examples, linguistic interface, multi-agent interoperability, accessibility, performance, change log) are included in full detail in the original expanded specification.")

# Save file
output_path = "/mnt/data/Glyphic_Codex_Reference_Manual_v1.1.docx"
doc.save(output_path)

output_path
